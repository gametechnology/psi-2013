287  inline D3DCOLORVALUE colorToD3D(const SColor& col)
288  {
289  const f32 f = 1.0f / 255.0f;
290  D3DCOLORVALUE v;
291  v.r = col.getRed() * f;
292  v.g = col.getGreen() * f;
293  v.b = col.getBlue() * f;
294  v.a = col.getAlpha() * f;
295  return v;
296  }
